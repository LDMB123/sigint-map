Fix #1 (10 min): Re-enable wasm-opt
  File:  wasm/dmb-force-simulation/Cargo.toml
  From:  wasm-opt = false
  To:    wasm-opt = ["-Oz", "--enable-bulk-memory"]

Fix #2 (5 min): Add LICENSE
  Create: /LICENSE file with MIT/Apache-2.0 license
  Update: wasm/Cargo.toml [workspace.package] license field

Fix #3 (15 min): Add compression scripts to package.json
  "wasm:compress": "for f in wasm/*/pkg/*_bg.wasm; do brotli -j -11 \"$f\" -o \"$f.br\"; gzip -9 \"$f\" -c > \"$f.gz\"; done"
  "wasm:size-total": "ls -lh wasm/*/pkg/*_bg.wasm* | awk '{print $5, $9}'"

Fix #4 (5 min): Add metrics script
  "wasm:stats": "echo '=== WASM Module Breakdown ===' && du -sh wasm/*/pkg/"

Fix #5 (20 min): Configure Vite compression
  Install:  npm install -D vite-plugin-compression
  Add to vite.config.ts:
    import compression from 'vite-plugin-compression';
    plugins: [
      compression({
        algorithm: 'brotli',
        ext: '.br',
        threshold: 10240,
        level: 11,
      }),
      // ... other plugins
    ]

Test:  npm run wasm:build && npm run wasm:compress
Verify: npm run wasm:size-total
