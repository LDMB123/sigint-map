## Steps

### Step 1: Identify Critical Routes

```typescript
// src/lib/sw/routes.ts
// Routes that MUST work offline
export const CRITICAL_ROUTES = [
  '/',              // Homepage
  '/about',         // About page
  '/offline',       // Offline fallback
];

// Routes that SHOULD work if cached
export const SECONDARY_ROUTES = [
  '/features',
  '/pricing',
  '/contact',
];

// Dynamic routes that can work offline with cached data
export const DYNAMIC_ROUTES = [
  '/blog/[slug]',
  '/products/[id]',
```

### Step 2: Implement Precaching

```javascript
// static/sw.js or via vite-plugin-pwa
const PRECACHE_VERSION = 'v1';
const PRECACHE_URLS = [
  '/',
  '/about',
  '/offline',
  // Add critical CSS/JS (auto-generated by build tools)
];

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(`precache-${PRECACHE_VERSION}`).then((cache) => {
      return cache.addAll(PRECACHE_URLS);
    })
  );
  self.skipWaiting();
});
```

### Step 3: Implement Network-First with Offline Fallback

```javascript
// static/sw.js
self.addEventListener('fetch', (event) => {
  // Only handle navigation requests
  if (event.request.mode === 'navigate') {
    event.respondWith(
      fetch(event.request)
        .then((response) => {
          // Clone response for caching
          const responseClone = response.clone();
          caches.open('pages-v1').then((cache) => {
            cache.put(event.request, responseClone);
          });
          return response;
        })
        .catch(() => {
          // Network failed, try cache
          return caches.match(event.request)
            .then((response) => {
              if (response) return response;
              // Fallback to offline page
              return caches.match('/offline');
            });
        })
    );
  }
});
```

### Step 4: Create Offline Page

```svelte
<!-- src/routes/offline/+page.svelte -->
<script>
  import { onMount } from 'svelte';

  let cachedPages = $state([]);
  let isOnline = $state(navigator.onLine);

  onMount(() => {
    // List cached pages
    caches.open('pages-v1').then((cache) => {
      cache.keys().then((keys) => {
        cachedPages = keys.map((k) => new URL(k.url).pathname);
      });
    });

    // Listen for online/offline events
    const handleOnline = () => { isOnline = true; };
    const handleOffline = () => { isOnline = false; };

    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);

    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  });
</script>

<svelte:head>
  <title>Offline - Your App</title>
</svelte:head>

<div class="offline-page">
  <h1>You're Offline</h1>

  {#if isOnline}
    <p class="online-notice">Connection restored! You can now browse normally.</p>
  {:else}
    <p>Some features are unavailable without internet.</p>
  {/if}

  {#if cachedPages.length > 0}
    <section>
      <h2>Available Offline:</h2>
      <ul class="cached-pages">
        {#each cachedPages as page}
          <li>
            <a href={page}>{page}</a>
          </li>
        {/each}
      </ul>
    </section>
  {/if}
</div>
```

### Step 5: Handle Data Loading with Offline Support

```svelte
<!-- src/routes/+page.svelte -->
<script>
  import { onMount } from 'svelte';

  let data = $state(null);
  let isOffline = $state(false);
  let error = $state(null);

  onMount(async () => {
    try {
      const response = await fetch('/api/data');
      if (!response.ok) throw new Error('Failed to fetch');
      data = await response.json();
    } catch (err) {
      // Try to load from cache or IndexedDB
      isOffline = true;
      try {
        data = await loadFromCache();
      } catch (cacheErr) {
        error = 'Unable to load data offline';
      }
    }
  });

  async function loadFromCache() {
    // Load from IndexedDB or Service Worker cache
    const cache = await caches.open('api-cache');
    const response = await cache.match('/api/data');
    if (response) {
      return response.json();
    }
    throw new Error('No cached data');
  }
</script>

{#if error}
  <div class="error">{error}</div>
{:else if data}
  <div class="content">
    {#if isOffline}
      <div class="offline-notice">Showing cached data</div>
    {/if}
    <!-- Render data -->
  </div>
{:else}
  <div class="loading">Loading...</div>
{/if}
```

## App Shell Pattern

### What to Precache (Shell)

```
├── index.html (or root route)
├── app.css (critical CSS)
├── app.js (critical JS)
├── navigation component
├── header/footer
└── loading skeleton
```

### What to Load On-Demand

```
├── Page-specific content
├── Images
├── API responses
├── User data
└── Non-critical features
```

## Caching Strategies by Content Type

### Static Pages (Homepage, About)
**Strategy**: Cache First
```javascript
{
  urlPattern: /^\/(about|contact|features)$/,
  handler: 'CacheFirst',
  options: {
    cacheName: 'static-pages',
    expiration: { maxAgeSeconds: 60 * 60 * 24 * 7 } // 1 week
  }
}
```

### Dynamic Content (Blog, Products)
**Strategy**: Network First
```javascript
{
  urlPattern: /^\/(blog|products)\/.*/,
  handler: 'NetworkFirst',
  options: {
    cacheName: 'dynamic-pages',
    networkTimeoutSeconds: 3,
    expiration: { maxEntries: 50 }
  }
}
```

### API Calls
**Strategy**: Network First with short timeout
```javascript
{
  urlPattern: /\/api\/.*/,
  handler: 'NetworkFirst',
  options: {
    cacheName: 'api-cache',
    networkTimeoutSeconds: 2,
    expiration: { maxAgeSeconds: 60 * 5 } // 5 minutes
  }
}
```
